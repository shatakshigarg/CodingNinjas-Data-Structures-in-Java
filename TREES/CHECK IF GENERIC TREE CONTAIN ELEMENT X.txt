	public static boolean checkIfContainsX(TreeNode<Integer> root, int x){

        while(!queue.isEmpty())
        {
            TreeNode<Integer> frontNode = queue.remove();
            if(frontNode == null)
            {
                if(queue.isEmpty()){
                    break;
                }
                
                queue.add(null);
            }
            else{
               			 if(frontNode.data==x)
           			 		return true;

                    queue.add(frontNode.children.get(i));
                }
            }
            
        }
        
        return false;
